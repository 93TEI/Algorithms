# 체육복

# 전체 학생 n, 도난당한 학생들 번호 배열 lost, 여벌 체육복 학생 번호 배열 reserve

import collections
def solution(n, lost, reserve):
    answer = n - len(lost)
    for i in lost : # lost의 앞번호 학생이 여분을 가지고 있다면 1 올리기
        temp = collections.Counter([i-1]) - (collections.Counter([i-1])-collections.Counter(reserve))
        if temp.get(i-1) != None :
            answer += 1
        else : # 앞번호 학생이 여분이 없다면 뒷번호 학생의 여분 확인해서 1 올리기
            temp = collections.Counter([i+1]) - (collections.Counter([i+1])-collections.Counter(reserve))
            if temp.get(i+1) != None :
                answer += 1
                reserve.pop(i+1) # 빌려주면 여분에서 빼기
    return answer
